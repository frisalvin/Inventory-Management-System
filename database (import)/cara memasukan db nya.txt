CREATE TABLE barang (
id INT AUTO_INCREMENT PRIMARY KEY,
kode_barang VARCHAR(50) UNIQUE,
nama_barang VARCHAR(100),
kategori VARCHAR(50),
satuan VARCHAR(20),
harga_modal INT DEFAULT 0,
keterangan TEXT,
stok_awal INT DEFAULT 0
);

CREATE TABLE stok_masuk (
id INT AUTO_INCREMENT PRIMARY KEY,
tanggal DATE,
kode_barang VARCHAR(50),
jumlah INT,
supplier VARCHAR(100),
keterangan TEXT
);

CREATE TABLE stok_keluar (
id INT AUTO_INCREMENT PRIMARY KEY,
tanggal DATE,
kode_barang VARCHAR(50),
jumlah INT,
penerima VARCHAR(100),
keterangan TEXT
);

CREATE TABLE stok_opname (
id INT AUTO_INCREMENT PRIMARY KEY,
tanggal DATE,
kode_barang VARCHAR(50),
stok_sistem INT,
stok_fisik INT,
selisih INT
);
